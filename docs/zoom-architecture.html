    subgraph USER["👤 End User (Browser)"]
        A[Client books appointment
on denik.me]
    end

    subgraph DENIK["🖥️ Denik Agenda Server
(Fly.io — Node.js / React Router v7)"]
        B[OAuth Callback Handler
/dash/zoom/callback]
        C[Event Creation API
/api/events]
        D[Zoom Service Layer
zoom.server.ts]
        E[Email Notification
AWS SES]
    end

    subgraph DB["🗄️ MongoDB Atlas"]
        F[(Org — stores Zoom tokens
zoomAccessToken
zoomRefreshToken
zoomExpiresAt)]
        G[(Event — stores meeting info
meetingLink, meetingId)]
    end

    subgraph ZOOM["☁️ Zoom API"]
        H[OAuth 2.0
zoom.us/oauth]
        I[Create Meeting
POST /v2/users/me/meetings]
        J[Delete Meeting
DELETE /v2/meetings/:id]
    end

    %% OAuth Flow
    H -- "1. Auth code" --> B
    B -- "2. Exchange code for tokens" --> H
    B -- "3. Store tokens" --> F

    %% Booking Flow
    A -- "4. Book appointment" --> C
    C -- "5. Create Zoom meeting" --> D
    D -- "6. Get/refresh tokens" --> F
    D -- "7. POST create meeting" --> I
    I -- "8. Meeting URL + ID" --> D
    D -- "9. Save meeting link" --> G
    C -- "10. Send confirmation with Zoom link" --> E
    E -- "11. Email with meeting link" --> A

    %% Cancellation
    C -. "Cancel: delete meeting" .-> D
    D -. "DELETE meeting" .-> J
